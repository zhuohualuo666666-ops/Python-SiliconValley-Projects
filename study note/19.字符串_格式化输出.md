<!-- 19.字符串_格式化输出 -->

格式化输出写法：
·写法1：加号拼接,只能在字符串之间进行拼接
例：
name = '张三'
gender = '男'
info1 = '我叫' + name + '，我是' + gender + '生'
print(info1)

·写法2：使用占位符，%s占位字符串，%f占位浮点数，%i占位浮点数，%d占位十进制整数，%s万能
name = '张三'
gender = '男'
age = 18
weight = 60.2
info2 = '我叫%s，我是%s生,我体重是%f,我年龄是%d' % (name, gender, weight, age)
print(info2)

·写法3：使用f-string，官方推荐
name = '张三'
gender = '男'
age = 18
weight = 60.2
info3=f'我叫{name},我是{gender}生,我体重是{weight},我年龄是{age}'
print(info3)
